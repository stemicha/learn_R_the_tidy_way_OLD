ggside package

The package ggside was designed to enable users to add metadata to their ggplots with ease. While adding metadata information is not extremely difficult to do with geom_tile or other geoms, it can be frustrating to the user positioning these geometries away from the main plot. Additionally, if the user wants to use a color guide with the fill aesthetic, then they may run into conflicts when one layer uses a discrete scale and another uses a continuous scale.

# install ggside
install.packages("ggside")
library(tidyverse)
library(ggside)

i2 <- iris %>%
  mutate(Species2 = rep(c("A","B"), 75))

ggplot(data = i2,mapping = aes(x = Sepal.Width,y = Sepal.Length, color = Species))+
  geom_point()+
  geom_ysidedensity(aes(x=stat(density)))+ # adding y-axis density plot
  geom_xsidedensity(aes(y=stat(density)))+ # adding x-axis density plot
  theme_light()

ggplot(data = i2,mapping = aes(x = Sepal.Length,y = Sepal.Width, color = Species))+
  geom_point()+
  geom_ysidedensity(aes(x=stat(density)))+ # adding y-axis density plot
  geom_xsidedensity(aes(y=stat(density)))+ # adding x-axis density plot
  theme_light()+
  stat_smooth(method = "lm")+#linear fit
  facet_grid(Species~Species2) + #adding facet_grid
  labs(title = "FacetGrid") +
  guides(guide_axis(check.overlap = T)) # removing redundant guides
## `geom_smooth()` using formula 'y ~ x'

ggplot(mpg, aes(displ, hwy, colour = class)) + 
  geom_point(size = 2) +
  geom_xsideboxplot(aes(y = class), orientation = "y") +
  geom_ysidedensity(aes(x = after_stat(density)), position = "stack") +
  theme(ggside.panel.scale = .3) +
  scale_xsidey_discrete() +
  scale_ysidex_continuous(guide = guide_axis(angle = 90), minor_breaks = NULL)+
  theme_light()+
  scale_color_brewer(palette = "Dark2")
